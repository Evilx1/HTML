VB script

                --- Declare variables using Dim
                Dim name, age
                
                ---Assign values using InputBox
                name = InputBox("Enter your name:")
                age = InputBox("Enter your age:")
                
                --- Display values using MsgBox
                MsgBox "Hello, " & name & "! You are " & age & " years old."
                
                ---Conditional statement using If Then Else
                If age >= 18 Then
                    MsgBox "You are an adult."
                Else
                    MsgBox "You are a minor."
                End If
